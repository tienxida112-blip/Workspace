    local container = Instance.new("Frame", Content)
    container.Size = UDim2.new(0.9, 0, 0, 55)
    container.BackgroundColor3 = Color3.fromRGB(22, 22, 22)
    CreateRoundedCorner(container, 8)

    label.Text = name .. ": " .. default
    label.Size = UDim2.new(1, -20, 0, 25)
    label.Position = UDim2.new(0, 10, 0, 5)
    label.TextColor3 = Color3.new(1, 1, 1)
    label.BackgroundTransparency = 1
    label.Font = Enum.Font.Gotham
    label.TextXAlignment = Enum.TextXAlignment.Left

    local bg = Instance.new("Frame", container)
    bg.Size = UDim2.new(0.9, 0, 0, 4)
    bg.Position = UDim2.new(0.05, 0, 0.75, 0)
    bg.BackgroundColor3 = Color3.new(0.1, 0.1, 0.1)
    CreateRoundedCorner(bg, 2)

    local fill = Instance.new("Frame", bg)
    fill.Size = UDim2.new((default - (isSmooth and 0 or min)) / (max - min), 0, 1, 0)
    fill.BackgroundColor3 = Color3.fromRGB(0, 180, 255)
    CreateRoundedCorner(fill, 2)

    local dot = Instance.new("Frame", fill)
    dot.Size = UDim2.new(0, 18, 0, 18)
    dot.Position = UDim2.new(1, -9, 0.5, -9)
    dot.BackgroundColor3 = Color3.new(1, 1, 1)
    CreateRoundedCorner(dot, 50)
    dot.Active = true

    local dragging = false
    local function update()
        local input = UserInputService:GetMouseLocation()
        local rel = math.clamp((input.X - bg.AbsolutePosition.X) / bg.AbsoluteSize.X, 0, 1)
        local val = min + (max - min) * rel
        
        if isSmooth then
            Settings.Smoothing = math.round(val * 100) / 100
            label.Text = name .. ": " .. Settings.Smoothing
            fill.Size = UDim2.new(rel, 0, 1, 0)
        else
            Settings.PredictAmount = math.floor(val)
            label.Text = name .. ": " .. Settings.PredictAmount .. " Studs"
            fill.Size = UDim2.new(rel, 0, 1, 0)
        end
    end

    dot.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then dragging = true end end)
    UserInputService.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then dragging = false end end)
    RunService.RenderStepped:Connect(function() if dragging then update() end end)
end

-- AIMBOT ENGINE
local function getClosestTarget()
    if not Settings.Enabled then return nil end
    local closest, maxDist = nil, math.huge
    local myRoot = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not myRoot then return nil end

    local function check(obj)
        local hum = obj:FindFirstChildOfClass("Humanoid")
        local root = obj:FindFirstChild("HumanoidRootPart")
        if hum and root and hum.Health > 0 then
            local dist = (root.Position - myRoot.Position).Magnitude
            if dist < maxDist then maxDist = dist; closest = obj end
        end
    end

    if Settings.TargetMode == "Player" or Settings.TargetMode == "All" then
        for _, p in pairs(Players:GetPlayers()) do if p ~= LocalPlayer and p.Character then check(p.Character) end end
    end
    if Settings.TargetMode == "Mob" or Settings.TargetMode == "All" then
        local folder = workspace:FindFirstChild("Enemies") or workspace
        for _, m in pairs(folder:GetDescendants()) do if m:IsA("Model") and m:FindFirstChild("Humanoid") and not Players:GetPlayerFromCharacter(m) then check(m) end end
    end
    return closest
end

RunService.RenderStepped:Connect(function()
    if not running or not Settings.Enabled then 
        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then LocalPlayer.Character.Humanoid.AutoRotate = true end
        return 
    end

    local holding = Settings.IsMobile and (Settings.Enabled) or false
    if not Settings.IsMobile then
        for _, k in pairs(Settings.ValidKeys) do if UserInputService:IsKeyDown(k) then holding = true break end end
        if UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) then holding = true end
    end

    if holding then
        local t = getClosestTarget()
        if t then lastTarget, isAiming, releaseTime = t, true, tick() end
    elseif tick() - releaseTime > Settings.AimDelay then
        isAiming = false
    end

    if isAiming and lastTarget and lastTarget:FindFirstChild("HumanoidRootPart") then
        local char = LocalPlayer.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChild("Humanoid")
        if root and hum then
            hum.AutoRotate = false
            local tRoot = lastTarget.HumanoidRootPart
            local predictedPos = tRoot.Position + (tRoot.Velocity * (Settings.PredictAmount / 15))
            Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, predictedPos), Settings.Smoothing)
            root.CFrame = root.CFrame:Lerp(CFrame.new(root.Position, Vector3.new(predictedPos.X, root.Position.Y, predictedPos.Z)), Settings.Smoothing)
        end
    elseif LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then
        LocalPlayer.Character.Humanoid.AutoRotate = true
    end
end)
